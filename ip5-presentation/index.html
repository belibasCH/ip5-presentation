Implementierung

             
              addCommand (event) {
                let newValue = event.value
                if (this.activity.activity_type === running && event.type === 'pace') {
                  newValue = this.reconvertPaceForSave(newValue)
                }
                if (event.type === 'speed') {
                  newValue = newValue / this.speedConstant
                }
                newValue = parseFloat(newValue)
                this.undoStack.push(
                  {
                    'identity': 'area_editor',
                    'type': event.type,
                    'value': newValue,
                    'start_time': this.formatTime(this.startEdit),
                    'end_time': this.formatTime(this.endEdit)
                  }
                )
              },

              undo () {
                const moveCommand = this.undoStack.pop()
                this.redoStack.push(moveCommand)
              },
              redo () {
                const moveCommand = this.redoStack.pop()
                this.undoStack.push(moveCommand)
              },
